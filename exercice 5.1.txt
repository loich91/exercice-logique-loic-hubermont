Ecrire un programme qui lit 30 nombres et affiche la moyenne des nombres compris 
entre 10 et 20.

déclaration moyenne-interval
    declaration variable
    const TAILLE:N <- 30
    var tab :N [TAILLE]
    var somme :N
    var moyenne :N
    var ind :N <- 1
    var nombres:N

    fin declaration variable
    afficher "vieillez saisir nombre"

tant que ind < TAILLE
    saisir nombres
    si nombres>=10 et nombres<=20
       alors tab[ind]=nombres
        ind <- ind + 1
    fin si
fin tant que

somme=tab[1]

pour ind  allant de 2 à TAILLE
    somme=somme+tab[ind]
suivant

moyenne = somme/TAILLE

afficher moyenne
fin declaration moyenne-interval
