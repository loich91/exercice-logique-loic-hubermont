Ecrire un programme qui lit 30 nombres et affiche la moyenne des nombres compris 
entre 10 et 20.
déclaration moyenne-interval-corrrigé
    declaration variable
    const TAILLE:N <- 30
    var tab :N [TAILLE]
    var somme :N <-0
    var moyenne :N
    var ind :N <- 1
    var nombres:N
    var nombresCase <-0

    fin declaration variable
    afficher "vieillez saisir nombre"
pour ind allant de 1 à TAILLE
    saisir nombres
    tab[ind]=nombres
ind suivant

pour ind allant de 1 à TAILLE
    si tab[ind]>=10 et tab[ind]<=20
       alors somme = somme+tab[ind]
            nombresCase <- nombresCase + 1
    fin si
ind suivant
si nombresCase = 0
alors afficher "erreur aucun des nombres rentré ne permet de faire la moyenne"
sinon
    moyenne=somme/nombresCase
    afficher moyenne
finsi

fin moyenne-interval-corrrigé
