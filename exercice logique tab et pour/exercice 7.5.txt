Supprimer espaces
Ecrire le programme qui déplace
les caractères espaces à la fin d’un vecteur de caractères.

declaration supprimer-espaces
    debut declaration variable
        const TAILLE:N <- 30
        var tab:T [TAILLE]
        var tab-resultat:T[TAILLE]
        var espaces:T <- " "
        var ind:N <- 1
        var nbreCaractere:N 
        var j:N <- 1
        var chaine:T <-""


    fin declaration variable
nombres-espaces <- 0;
nbreCaractere <- 0;
pour ind allant de 1 à TAILLE
    si tab[ind] <> espaces
        alors
            chaine <- chaine + tab[ind]
    finsi
suivant
nbreCaractere <- longueur(chaine)
j<- 1

pour ind allant de 1 à TAILLE
    si ind <= nbreCaractere
        alors tab-resultat[j] <- caract(chaine,j)
        j <- j +1
        sinon 
            tab-resultat[j] <- espace
            j <- j + 1

suivant

fin  declaration supprimer-espaces
