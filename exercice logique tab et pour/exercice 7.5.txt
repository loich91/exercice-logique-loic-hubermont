Supprimer espaces
Ecrire le programme qui déplace
les caractères espaces à la fin d’un vecteur de caractères.

declaration supprimer-espaces
    debut declaration variable
        const TAILLE:N <- 30
        var tab:T [TAILLE]
        var tab-resultat:T[TAILLE]
        var espaces:T <- " ";
        var nombres-espaces:N
        var ind:N <- 1
        var nbreCaractere:N 

    fin declaration variable
nombres-espaces <- 0;
nbreCaractere <- 0;
pour ind allant de 1 à TAILLE
    si tab[ind] = espaces
    alors
       nombres-espaces <- nombres-espaces + 1
    finsi
suivant
nbreCaractere <- TAILLE - nombres-espaces
pour ind allant de 1 à TAILLE
    si tab[ind] <> espaces
        alors tab-resultat<-tab[ind]
    finsi
    si ind > nbreCaractere
        alors tab-resultat[ind] <- espaces
    finsi
suivant
fin  declaration supprimer-espaces
