entré tableau trié rechercher une valeur voir si elle est présente et la position



function recherche(in tab[]:N,TAILLE:N, in nbre:N, out position:N):N
var ind <-1
var position <- -1
    tant que ind < TAILLE et position = -1
        si tab[ind] = nbre
            alors position <- ind
        finsi
    ind suivant 
fin recherche
